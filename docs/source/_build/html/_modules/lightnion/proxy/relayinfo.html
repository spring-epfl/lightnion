[docs]def get_relay_info(port, password=None, params=None):
    """Retrieve some info from the relay.
    """

    if params is None:
        params = DEFAULT_PARAMS

    with Controller.from_port(port=port) as controller:
      controller.authenticate(password=password)

      info = controller.get_info(params)

      return info

    raise Exception('An error occurred while retrieving the info from the relay.')
